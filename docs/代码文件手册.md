#### 2.1 supervisor.py - 主 Supervisor Agent

**职责**: 
- 接收用户请求
- 分解复杂任务
- 派发子 Agent
- 汇总结果

**核心类**:

```python
class SupervisorAgent:
    """主监督 Agent"""
    
    def __init__(self, settings: Settings):
        self.llm = create_llm(settings)
        self.middleware_manager = MiddlewareManager()
        self.thread_states = {}  # 线程状态存储
    
    async def invoke(
        self,
        message: str,
        thread_id: str = "default",
        context: Optional[Dict] = None
    ) -> Dict[str, Any]:
        """执行 Agent"""
        # 1. 初始化/读取线程状态
        # 2. 执行 before_agent 钩子
        # 3. 准备运行时工具
        # 4. 调用 LLM 和工具
        # 5. 执行 after_agent 钩子
        # 6. 返回结果
```

| `_to_runtime_messages()` | 消息格式转换 |

## 中间件系统

### 系统架构

```
用户请求
    ↓
[before_agent] ──→ 状态初始化检查
    ↓
[before_model] ──→ 工具加载、Schema 加载
    ↓
[wrap_model_call] ──→ 模型调用包装
    ↓
[after_model] ──→ 响应分析
    ↓
[wrap_tool_call] ──→ 权限检查 → 缓存查询 → 上下文编辑 → 工具执行
         ├──→ ToolAuthMiddleware (priority=80)
         ├──→ ToolCacheMiddleware (priority=50)
         └──→ ContextEditingMiddleware (priority=10)
    ↓
[after_tool_call] ──→ 结果处理
    ↓
[after_agent] ──→ 清理、持久化
    ↓
返回结果
```

### 中间件类型

#### 1. middleware/base.py - 基础架构

```python
class BaseMiddleware(ABC):
    """中间件基类"""
    name: str
    priority: int  # 优先级（越大越先执行）
    
    async def before_agent(self, state: Dict) -> Optional[MiddlewareCommand]:
        """Agent 启动前"""
        pass
    
    async def wrap_tool_call(
        self,
        state: Dict,
        tool_call: Any,
        handler: Callable
    ) -> Any:
        """包装工具调用（支持链式组合）"""
        pass

class MiddlewareManager:
    """中间件管理器"""
    
    def add_middleware(self, middleware: BaseMiddleware):
        """添加中间件（自动按优先级排序）"""
    
    async def run_wrap_tool_call(
        self,
        state: Dict,
        tool_call: Any,
        handler: Callable
    ) -> Any:
        """执行工具调用中间件链"""
```

#### 2. middleware/permissions.py - RBAC 权限

```python
class Role(Enum):

    ANALYST = "analyst"   # 读写+执行权限
    ADMIN = "admin"       # 全部权限

class PermissionManager:
    """权限管理器"""
    
    def check(self, role: Role, tool_name: str) -> bool:
        """检查权限"""
    
    def register_tool_permission(
        self,
        tool_name: str,
        permission: ToolPermission
    ):
        """注册工具权限"""
```

#### 3. middleware/tool_auth.py - 认证中间件

```python
class ToolAuthMiddleware(BaseMiddleware):
    """工具权限认证中间件"""
    
    name = "tool_auth"
    priority = 80  # 最高优先级，在缓存前执行
    
    async def wrap_tool_call(
        self,
        state: Dict,
        tool_call: Any,
        handler: Callable
    ) -> Any:
        """检查工具权限"""
        # 1. 提取用户角色
        # 2. 调用 PermissionManager.check()
        # 3. 如果权限不足，返回错误
        # 4. 否则调用 handler()
```

#### 4. middleware/tool_cache.py - 缓存中间件

```python
class ToolCacheMiddleware(BaseMiddleware):
    """工具调用缓存中间件"""
    
    name = "tool_cache"
    priority = 50  # 在认证后、上下文编辑前
    
    async def wrap_tool_call(
        self,
        state: Dict,
        tool_call: Any,
        handler: Callable
    ) -> Any:
        """缓存工具调用结果"""
        # 1. 跳过写类工具（write_file, delete_file 等）
        # 2. 计算缓存 key (SHA256(tool_name:args))
        # 3. 如果缓存命中，返回 [缓存] 标记结果
        # 4. 否则执行工具，存入缓存
        # 5. 更新命中率统计
```

#### SQL 工具

```python
# tools/sql/query_tool.py → 工具名: sql_inter

class SQLQueryTool(BaseCustomTool):
    name = "sql_inter"
    description = "在 MySQL 数据库中执行 SQL 查询（仅 SELECT）"
    args_schema = SQLQueryInput  # 唯一参数：sql_query: str
    
    def _execute(self, sql_query: str) -> ToolResult:
        """执行 SQL"""
        # 安全检查：_is_safe_query() 拦截 INSERT/UPDATE/DELETE/DROP 等
        try:
            with DatabaseManager() as db:
                results = db.execute_query(sql_query)
                return ToolResult(
                    success=True,
                    data=results,
                    metadata={"rows": len(results)}
                )
        except Exception as e:
            return ToolResult(success=False, error=str(e))
    # 结果超过 1000 条自动截断；返回 str 经 to_tool_message_content() 转换

# 自动注册
sql_inter = SQLQueryTool()
registry.register(sql_inter)
```

```python
class DatabaseManager:
    """数据库连接管理器"""
    
    def __init__(self, settings: DatabaseSettings):
        self.pool = create_connection_pool(settings)
    
    def __enter__(self) -> Database:
        """上下文管理器入口"""
        return self.pool.get_connection()
    
    def execute_query(self, sql: str, params: Tuple = ()) -> List[Dict]:
        """执行查询"""
    
    def execute_update(self, sql: str, params: Tuple = ()) -> int:
        """执行更新"""
```

```python
    thread_id: str = "default"
    context: Optional[Dict] = None

class ChatResponse(BaseModel):
    messages: List[Dict]
    status: str
    thread_id: str

@router.post("/chat", response_model=ChatResponse)
async def chat_endpoint(request: ChatRequest):
    """聊天端点"""
    agent = get_supervisor_agent()
    result = await agent.invoke(
        message=request.message,
        thread_id=request.thread_id,
        context=request.context
    )
    return ChatResponse(**result)
```

```python
# middleware/my_middleware.py

from middleware.base import BaseMiddleware, MiddlewareCommand

class MyMiddleware(BaseMiddleware):
    name = "my_middleware"
    priority = 30  # 中等优先级
    
    async def before_agent(self, state: Dict[str, Any]) -> Optional[MiddlewareCommand]:
        """Agent 启动前逻辑"""
        # 做一些初始化检查
        return None
    
    async def wrap_tool_call(
        self,
        state: Dict[str, Any],
        tool_call: Any,
        handler: Callable
    ) -> Any:
        """为工具调用添加额外行为"""
        # 1. 前置处理
        result = await handler(tool_call)
        # 2. 后置处理
        return result
    
    async def after_agent(
        self,
        state: Dict[str, Any],
        final_result: Any
    ) -> Optional[MiddlewareCommand]:
        """Agent 结束后清理"""
        return None
```

## 关键设计模式

### 中间件链模式

```
优先级: 80 (ToolAuth) → 50 (Cache) → 30 (Custom) → 10 (ContextEdit)

每个中间件通过 async handler 和 await 实现链式调用
```

### 工具注册模式

```
工具类定义 → 实例化 → registry.register() → 全局可用
```

### 状态驱动模式

```
线程状态 (thread_id) → 会话记忆 → 跨轮对话支持
```

### 仓库模式

```
DatabaseManager → Repository → 业务逻辑
```
